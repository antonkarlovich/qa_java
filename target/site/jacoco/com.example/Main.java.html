package com.example;

public class Main {
    public static void main(String[] args) throws Exception {
        Animal animal = new Animal();
        System.out.println(animal.getFamily());

        System.out.println(animal.getFood("Травоядное"));
        System.out.println(animal.getFood("Хищник"));
        System.out.println("--------------------------------------------");

        Feline feline = new Feline();
        Cat cat = new Cat(feline);
        System.out.println(cat.getFood());
        System.out.println(cat.getSound());

        System.out.println("---------------------------------------------");

        System.out.println(feline.eatMeat());
        System.out.println(feline.getKittens());
        System.out.println(feline.getKittens(2));
        System.out.println(feline.getFamily());

        System.out.println("---------------------------------------------");

    }
}
